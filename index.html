import matplotlib.pyplot as plt import numpy as np from matplotlib.animation import FuncAnimation from pyscript import display # PyScript 特有，用于显示图像在网页 # 创建画布 fig, ax = plt.subplots() ax.set_xlim(-150, 150) ax.set_ylim(-150, 150) points, = ax.plot([], [], 'o', color='red') # 初始化函数 def init(): points.set_data([], []) return points, # 每一帧动画更新函数 def animate(i): r = np.random.rand(20) * 100 theta = np.random.rand(20) * 2 * np.pi x = r * np.cos(theta) y = r * np.sin(theta) points.set_data(x, y) return points, # 创建动画 ani = FuncAnimation(fig, animate, init_func=init, frames=50, interval=100) # 显示图像在网页上（关键点） display(fig)
